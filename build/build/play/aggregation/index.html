Yaml

    
  select:
  - department_name
  - operator: min
    alias: min_salary
    args:
    - salary
  from: employees
  join:
  - from: departments
    type: inner_join
    on:
    - operator: =
      left: department_name
      right: "test"
  group:
  - department_name

SQL

SELECT department_name, min(salary) AS min_salary FROM "employees" INNER JOIN "departments" ON department_name = test GROUP BY department_name
