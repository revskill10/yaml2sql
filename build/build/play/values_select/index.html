WITH "countries" AS (SELECT * FROM (VALUES ('us', 'test1'), ('fr', 'France')) AS codes (code, name)) SELECT DISTINCT ON ( data.code ) * FROM "data" LEFT JOIN "countries" ON countries.code = data.code
